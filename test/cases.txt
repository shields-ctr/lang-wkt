# Point
POINT( 10 10 )
// and then a comment...
==> WKT(Geometry(Point(2D(X(Number),Y(Number)))), Comment)

#Point 3D
POINT Z (1 2 3)
==> WKT(
  Geometry(
    Point2( 3D( X(Number),Y(Number),Z(Number)))))

# LineString
LINESTRING(30 10, 10 30, 40 40)
==> WKT(
  Geometry(
    LineString(
      List(
        2D( X(Number), Y(Number) ),
        2D( X(Number), Y(Number) ),
        2D( X(Number), Y(Number) )
      )
    )
  )
)

# Polygon
POLYGON ((30 10, 40 40, 20 40, 10 20, 30 10))
==> WKT(
  Geometry(
    Polygon(
      List(
        2D( X(Number), Y(Number) ),
        2D( X(Number), Y(Number) ),
        2D( X(Number), Y(Number) ),
        2D( X(Number), Y(Number) ),
        2D( X(Number), Y(Number) )
      )
    )
  )
)

# MultiPoint
MULTIPOINT ((10 40), (40 30), (20 20), (30 10))
==> WKT(
  Geometry(
    MultiPoint(
      2D( X(Number), Y(Number) ),
      2D( X(Number), Y(Number) ),
      2D( X(Number), Y(Number) ),
      2D( X(Number), Y(Number) )
    )
  )
)
